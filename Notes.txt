for each ray
	-find angle of ray
		spacing = fov / dispW
		start ang = player angle + fov / 2
		ray ang = start ang - spacing / 2 - spacing * ray#
		
	-find distance to wall
		right wall
			distWall = (1-x)/cos(rayTheta)
		top wall
			distWall = (1-y)/cos(rayTheta)
		
	-adjust dist to perp from player angle?
		cos(pTheta-rayTheta) * distWall
		
	-determine height of column to display
		distMax = 2 // colH = 0
		distMin = 0.5 // colH = dispH
		colH = (distMax - distAdj) * (dispH / (distMax - distMin))
		
movement
	-forward/backward n units
		x +/-= n * cos(theta)
		y +/-= n * sin(theta)
	-left n units
		x -= n * sin(theta)
		y += n * cos(theta)
	-right n units
		x += n * sin(theta)
		y -= n * cos(theta)
